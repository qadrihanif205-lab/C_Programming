#include <stdio.h>

int main() {
    float physics, chemistry, mathematics, average;

    // Get student's PCM marks from the user
    printf("Enter marks in Physics: ");
    scanf("%f", &physics);
    printf("Enter marks in Chemistry: ");
    scanf("%f", &chemistry);
    printf("Enter marks in Mathematics: ");
    scanf("%f", &mathematics);

    // Calculate the average percentage in PCM
    average = (physics + chemistry + mathematics) / 3;

    // Check eligibility for admission using conditional operator
    (average >= 50) ? printf("Student is eligible for admission. Average PCM percentage: %.2f%%\n", average) : printf("Student is not eligible for admission. Average PCM percentage: %.2f%%\n", average);

    return 0;
}
